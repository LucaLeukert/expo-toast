## Install

```bash
bun add expo-toast
```

## Basic usage

```ts
import { toast } from 'expo-toast';

toast.success('Profile saved');

const id = toast.loading('Syncing...');
setTimeout(() => {
  toast.update(id, { variant: 'success', message: 'Done', duration: 'short' });
}, 1200);
```

## Promise flow

```ts
await toast.promise(uploadFile(), {
  loading: 'Uploadingâ€¦',
  success: 'Upload complete',
  error: (err) => `Upload failed: ${String(err)}`,
});
```

## Set defaults

```ts
toast.configure({
  position: 'top',
  size: 'fit-content',
  dedupeWindowMs: 1200,
  maxVisible: 3,
  maxQueue: 20,
  dropPolicy: 'oldest',
  motion: 'system',
});
```
